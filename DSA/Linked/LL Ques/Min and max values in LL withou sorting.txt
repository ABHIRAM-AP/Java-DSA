struct Node{
    int data;
    struct Node *next;
};

struct Node *GetNode();
struct Node *GetNode()
{
    struct Node *N=(struct Node *)malloc(sizeof(struct Node));
    return N;
}

struct Node *create(int);
struct Node *create(int n)
{
    struct Node *head=NULL;
    struct Node *temp;
    struct Node *newNode;
    
    int data,i=0;
    
    for(i=0;i<n;i++)
    {
        printf("Enter data:");
        scanf("%d",&data);
        
        newNode=GetNode();
        if(newNode==NULL)
        {
            printf("failed\n");
        }
        
        newNode->data=data;
        newNode->next=NULL;
        
        
        
        if(head==NULL){
            head=newNode;
        }
        else
        {
            temp->next=newNode;
        }
         temp=newNode;
    }
    return head;
}

void findMinMax(struct Node *head, int *min, int *max) {
    if (head == NULL) return;  // Handle empty list

    *min = head->data;  // Initialize min and max with the first node's data
    *max = head->data;

    struct Node *current = head->next;  // Start from the second node

    while (current != NULL) {
        if (current->data < *min) {
            *min = current->data;
        }
        if (current->data > *max) {
            *max = current->data;
        }
        current = current->next;
    }
}

    head = create(n);
findMinMax(head,&min,&max);
    printf("Minimum value is %d\n",min);
    printf("Maximum value is %d",max);
