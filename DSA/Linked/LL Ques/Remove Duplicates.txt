struct Node{
    int data;
    struct Node *next;
};

struct Node *GetNode();
struct Node *GetNode()
{
    struct Node *N=(struct Node *)malloc(sizeof(struct Node));
    return N;
}

struct Node *create(int);
struct Node *create(int n)
{
    struct Node *head=NULL;
    struct Node *temp;
    struct Node *newNode;
    
    int data,i=0;
    
    for(i=0;i<n;i++)
    {
        printf("Enter data:");
        scanf("%d",&data);
        
        newNode=GetNode();
        if(newNode==NULL)
        {
            printf("failed\n");
        }
        
        newNode->data=data;
        newNode->next=NULL;
        
        
        
        if(head==NULL){
            head=newNode;
        }
        else
        {
            temp->next=newNode;
        }
         temp=newNode;
    }
    return head;
}

void display(struct Node *head) {
    struct Node *temp = head;
    while (temp != NULL) {
        printf("%d\t", temp->data);
        temp = temp->next;
    }
}

struct Node *removeDup(struct Node*);
struct Node *removeDup(struct Node*head)
{
    struct Node *p=head;
    struct Node *temp=p->next;
    
    while(temp!=NULL)
    {
        if(temp->data== p->data)
        {
            temp=temp->next;
            continue;
        }
        
        p->next=temp;
        p=temp;
        temp=temp->next;
    }
    p->next=NULL;
    return head;
}
