#include <stdio.h>

#define MAXSIZE 10

typedef struct LL
{
    int data;
    struct LL *next; // Use 'struct LL' for the pointer reference to the next node
} LL;

typedef struct
{
    int top;
    int a[MAXSIZE];
}STACK;

void initialise(STACK *);
void initialise(STACK *S)
{
    S->top =-1;
}

void pop(STACK*);
void pop(STACK*S)
{
    int item;
    if(S->top ==-1)
    {
        printf("Stack Underflow\n");
    }
    else
    {
        item = S->a[S->top];
        printf("%d\t",item);
        S->top--;
    }
}




int main()
{
    LL A, B, C, *start;
    
    STACK S;    
    initialise(&S);
    start = &A; 
    
   
    A.data = 10;
    B.data = 20;
    C.data = 30;
    
   
    A.next = &B;
    B.next = &C;
    C.next = NULL;

    LL *temp = start;
    while (temp != NULL)
    {
        S.top++;
        S.a[S.top]= temp->data;
        temp = temp->next;
    }
    
    while(S.top!=-1)
    {
        pop(&S);
    }
    
    
    return 0;
}
